#!/bin/bash
g++ test4.c -o test4
echo "Running test 4"
./test4
sleep 1
i=1
prv_line=""
while read line; do 
if (($i != 1 && $i < 51 )); then
	if (( $prv_line > $line )); then
		echo "process do not run in ascending pid order"
		exit
	fi
fi
if (( $i == 51 )); then
	if (( $prv_line < $line )); then
		echo "waiting process do not count in smallest pid count"
		exit
	fi
fi
if (( $i > 52 && $i < 153)); then
	if [[ $line != "better"* ]]; then
		echo "does not run real time/better prio process before SC when needed"
		exit
	fi
fi
if (( $i > 154 )); then
	if (( $prv_line > $line )); then
		echo "process do not run in ascending pid order"
		exit
	fi
fi
i=$i+1
prv_line=$line
done < test4.txt
echo "Test 4 passed"
rm test4
rm test4.txt
