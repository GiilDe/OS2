#!/bin/bash
g++ test3.c -o test3
echo "Running test 3"
./test3
sleep 1
i=1
prv_line=""
while read line; do 
if (($i != 1 )); then
	if (( $prv_line < $line)); then
		echo "waiting process do not count in smallest pid count"
		exit
	fi
fi
i=$i+1
prv_line=$line
done < test3.txt
echo "Test 3 passed"
rm test3.txt
rm test3
