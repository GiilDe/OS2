#!/bin/bash
g++ test6.c -o test6
echo "Running test 6"
./test6
sleep 1
i=1
prv_line=""
while read line; do 
if (($i != 1 )); then
	if (( $prv_line > $line)); then
		echo "process do not run in ascending pid order when streesed"
		exit
	fi
fi
i=$i+1
prv_line=$line
done < test6.txt
echo "Test 6 passed"
rm test6.txt
rm test6
